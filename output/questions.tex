\begin{enumerate}%
\item%
Let $\mathbf{F}$ be the vector field $\mathbf{F}(x, y, z)=z\mathbf{\hat{i}}+\mathbf{\hat{j}}+x^{2} y\mathbf{\hat{k}}$ and $C$ the curve given by $\mathbf{r}(t)=2 t\mathbf{\hat{i}}+2 t \left(t + 1\right)\mathbf{\hat{j}}+t^{2}\mathbf{\hat{k}}$ for $0\le t\le 1$. Calculate $\displaystyle\int_C\mathbf{F}\cdot\mathbf{dr}$.%
\item%
Let $\mathbf{F}$ be the vector field $\mathbf{F}(x, y, z)=- 2 y\mathbf{\hat{i}}+2 y z^{2}\mathbf{\hat{j}}+4\mathbf{\hat{k}}$ and $C$ the curve given by $\mathbf{r}(t)=-2\mathbf{\hat{i}}+\left(2 t + 1\right)\mathbf{\hat{j}}+\mathbf{\hat{k}}$ for $0\le t\le 3$. Calculate $\displaystyle\int_C\mathbf{F}\cdot\mathbf{dr}$.%
\item%
Let $\mathbf{F}$ be the vector field $\mathbf{F}(x, y, z)=- x \left(y + 2\right)\mathbf{\hat{i}}+x\mathbf{\hat{j}}-3\mathbf{\hat{k}}$ and $C$ the curve given by $\mathbf{r}(t)=-\mathbf{\hat{i}}+t\mathbf{\hat{j}}+\left(3 t^{3} + 1\right)\mathbf{\hat{k}}$ for $-2\le t\le 0$. Calculate $\displaystyle\int_C\mathbf{F}\cdot\mathbf{dr}$.%
\item%
Let $\mathbf{F}$ be the vector field $\mathbf{F}(x, y, z)=y\mathbf{\hat{i}}+3 \left(1 - x\right)\mathbf{\hat{j}}+\mathbf{\hat{k}}$ and $C$ the curve given by $\mathbf{r}(t)=t^{2} \left(t + 3\right)\mathbf{\hat{i}}+2\mathbf{\hat{j}}- t\mathbf{\hat{k}}$ for $-1\le t\le 1$. Calculate $\displaystyle\int_C\mathbf{F}\cdot\mathbf{dr}$.%
\end{enumerate}%
